cmake_minimum_required(VERSION 3.10.0)

project(open_drive_analyzer)

if (NOT TARGET adtf::ui)
    find_package(ADTF COMPONENTS filtersdk ui)
endif()

adtf_use_qt(Widgets Gui Core)

aux_source_directory(../Helpers HELPERS)
aux_source_directory(../Helpers/OpenDriveAnalyzer ODAnalyzer)
aux_source_directory(../Helpers/OpenDriveAnalyzer/mapReader ODMapReader)

set (SOURCEFILES

    ${HELPERS}
    ${ODAnalyzer}
    ${ODMapReader}

    MapVisualization.h
    MapVisualization.cpp
    displaywidget.cpp
    displaywidget.h
    stdafx.h

)



adtf_add_filter(${PROJECT_NAME}
                ${SOURCEFILES}
)

target_link_libraries(${PROJECT_NAME} PUBLIC adtf::ui Qt5::Widgets)


adtf_disable_qt_warnings(${PROJECT_NAME})

adtf_install_target(${PROJECT_NAME} bin)

#this is to add a folder within the ADTF examples solution
set_property(TARGET ${PROJECT_NAME} PROPERTY FOLDER position)

adtf_create_plugindescription(
    TARGET
        ${PROJECT_NAME}
    PLUGIN_SUBDIR
        "bin"
    DEPENDENT_PLUGINS
        "adtf_xsystem"
)
